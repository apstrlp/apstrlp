\section{Questions à choix multiple \hfill 6 pts,   \Clocklogo 20'}
%http://www.actionprojet.com/pageF11.htm
Pour chacune des questions suivantes, cochez la  bonne réponse.


\textbf{Remarque :} -0,5 pour chaque mauvaise réponse.

 \begin{enumerate}
 


\item \textbf{Parmi les qualificatifs suivants, cochez celui qui est incompatible avec la notion de projet :}

\begin{minipage}{5cm}
\begin{enumerate}[a. $\Box$ ]
\item original 
\item autonome
\item permanent   %%%
 \end{enumerate}
\end{minipage}
\hspace{5mm}
\begin{minipage}{5cm}
\begin{enumerate}

\item [d. $\Box$ ] novateur
\item [e. $\Box$ ] complexe 
\item [f. $\Box$ ] évolutif
 \end{enumerate}
\end{minipage}


\vfill %\vspace{2mm} %----------------------------------------------------------------------
%\rule{.95\textwidth}{1pt}

\item \textbf{Au sens du management de projet, on appelle ouvrage :}


\begin{enumerate}[a. $\Box$ ]
\item l'ensemble des actions réalisées au cours du projet 
\item la méthode utilisée pour réaliser le projet 
\item un élément matériel constitutif du projet 
\item l'objet physique ou intellectuel du projet %%%
\end{enumerate}

\vfill %\vspace{2mm} %----------------------------------------------------------------------
%\rule{.95\textwidth}{1pt}


\item \textbf{Dans la réalisation d'un projet, le maître d'ouvrage est celui qui :}
\begin{enumerate}[a. $\Box$ ]
\item définit le cahier des charges d'ouvrage %%%
\item est responsable de la coordination des intervenants 
\item  assure les études de conception 
\item dirige le chantier de construction de l'ouvrage

\end{enumerate}




\vfill %\vspace{2mm} %----------------------------------------------------------------------
%\rule{.95\textwidth}{1pt}


\item  \textbf{Durant l'exécution d'un projet, le chef de projet peut éventuellement se décharger de certaines tâches : }


\begin{enumerate}[a. $\Box$ ]
\item définition des objectifs à court terme de l'équipe 
\item définition des solutions techniques
\item  analyse des dérives de coûts 
\item  analyse des dérives de délai 
\item  analyse des dérives de qualité
\end{enumerate}


\vfill %\vspace{2mm} %----------------------------------------------------------------------
%\rule{.95\textwidth}{1pt}

\item \textbf{La fonction « chef de projet » est une charge : }

\begin{enumerate}[a. $\Box$ ]
\item  caractéristique du maître d'ouvrage 
\item caractéristique du maître d'oeuvre 
\item caractéristique d'un ensemble d'outils 
\item nécessaire chez tous les intervenants

\end{enumerate}

\vfill %\vspace{2mm} %----------------------------------------------------------------------
%\rule{.95\textwidth}{1pt}



\item \textbf{La réalisation d'un projet commence par la planification. La première étape de la planification consiste à :}


\begin{enumerate}[a. $\Box$ ]
\item Établir le diagramme de Gantt
\item Définir les jalons du projet
\item Établir l'organigramme des tâches %%%
\item Tracer la logique d'enchaînement de tâches

\end{enumerate}
\vfill %\vspace{2mm} %----------------------------------------------------------------------
%\rule{.95\textwidth}{1pt}

\newpage

\item \textbf{ Une politique de qualité comporte trois volets, lesquels : }


\begin{enumerate}[a. $\Box$ ]
\item technique, commercial, administratif 
\item conception, fabrication, maintenance 
\item planification, contrôle, amélioration
\item étude de marché, innovation, commercial  
\end{enumerate}

\vfill %\vspace{2mm} %----------------------------------------------------------------------
%\rule{.95\textwidth}{1pt}




\item \textbf{Les jalons (en anglais : milestones) :}

\begin{enumerate}[a. $\Box$ ]
\item sont importantes pour la communication et doivent de ce fait figurer sur le
GANTT.
\item ne font pas partie des préoccupations principales du chef de projet.
\item concernent les ressources affectées au projet.

\end{enumerate}

\vfill %\vspace{2mm} %----------------------------------------------------------------------
%\rule{.95\textwidth}{1pt}



\item \textbf{Pour quels objectifs l'estimation d'un budget n'est pas importante : }
\begin{enumerate}[a. $\Box$ ]
\item Demander des financements
\item Faire l'étude de marché
\item Rendre des comptes
\item \'Evaluer la faisabilité du projet
\item Suivre l'avancement du projet

\end{enumerate}

\vfill %\vspace{2mm} %----------------------------------------------------------------------
%\rule{.95\textwidth}{1pt}


\item \textbf{Qu'est-ce qu'un risque dans la gestion d'un projet?}

\begin{enumerate}[a. $\Box$ ]
\item   un événement incertain dont la concrétisation est catastrophique.   %%
\item  un événement complètement imprévisible.
\item  un événement du climat.
\item  quelque chose de dangereux.
\item  un événement pas complètement prévisible qui peut contredire les planification du projet. %%%

\end{enumerate}


\vfill %\vspace{2mm} %----------------------------------------------------------------------
%\rule{.95\textwidth}{1pt}


\item \textbf{Face à un risque très fréquent et très grave, que feriez vous ?}

\begin{enumerate}[a. $\Box$ ]
\item  Mettre un plan d'action pour réduire sa fréquence (prévention).
\item  Mettre un plan d'action pour réduire sa gravité (protection).
\item  Abandonner et/ou changer le projet.
\item  Mettre un plan d'action pour réduire à la fois sa fréquence et sa gravité.
\item  Appliquer la loi de Brooks.
\item  Démissionner ou changer d'entreprise.

\end{enumerate}
\vfill %\vspace{2mm} %----------------------------------------------------------------------
%\rule{.95\textwidth}{1pt}


\item \textbf{Une personne dans l'équipe perturbe les autres. Vous êtes ScrumMaster, comment réagissez-vous ? }%http://www.aubryconseil.com/post/Quizz-la-question-3-sur-le-ScrumMaster

\begin{enumerate}[a. $\Box$ ]
\item Vous la virez de l'équipe, la majorité est d'accord.
\item Vous ne faites rien, il n'y a pas de chef avec Scrum.
\item Vous l'invitez à prendre un café en dehors de l'entreprise. %%
\item Vous faites un rapport à la direction.
\item Faire comme si de rien n'était.

\end{enumerate}






\end{enumerate}

\subsection{Diagramme d'activité}
Donnez le diagramme d'activités correspondant au cas d'utilisation suivant :

\begin{itemize}
\item \textbf{Identification du cas d'utilisation : }
\begin{description}
\item [Nom :] commander un article 
\end{description}
\item \textbf{Fonctionnement du cas d'utilisation :} 
\begin{enumerate}[ ]
\item \textbf{Pré-condition :}
\begin{itemize} 
\item Le client possède une carte qu'il a achetée au magasin ; 
\item Le distributeur est alimenté en dvd. 
\end{itemize}
\item \textbf{Scénario :} 

\begin{itemize}
\item \textbf{Scénario nominal : }

\begin{enumerate}[a)]
\item  Le responsable de boutique vérifie la disponibilité de l'article ; 
\item  Le système de gestion de stock locale cherche les instances de l'article ; 
\item  Le responsable de boutique fait payer l'article ; 
\item  Le responsable de boutique facture le paiement et envoie la facture au système de la 
direction financière ; 
\item  La direction financière archive la facture. 
\end{enumerate}
\item \textbf{ scénarios alternatifs : }
\begin{enumerate}[ ]
\item [b-1)] Aucune instance de l'article n'est disponible : 
\item [b-1a)] Le système crée et archive une réservation et commande l'article au système 
 de gestion de stocks globale ; 
\item [b-1b)] Le système de gestion de stock globale enregistre la commande ; 
\item  [b-1c)] Le système de gestion de stock globale envoie l'article ; 
\item  [b-1d)] Le système de gestion de locale enregistre l'article [exception 1 : article non  retiré]. 
\item [c-1)] Si l'achat de l'article est t fait dans le cadre d'une réservation, rendre la réservation 
 inactive. 
\end{enumerate}
\item \textbf{Scénario d'exception :} 
\begin{enumerate}[]
\item \textbf{[exception 1]} Si l'article n'est pas retiré avant une semaine, la réservation est annulée. 
\end{enumerate}

\end{itemize}
\end{enumerate}
\end{itemize}
